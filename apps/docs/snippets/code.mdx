export const CodeSandbox = ({ files, height, previewHeight, showTabs }) => {
  const Element = 'code-sandbox';

  return (
    <Element
      class="not-prose"
      files={JSON.stringify(files)}
      height={height}
      style={previewHeight ? {'--preview-height': `${previewHeight}px`} : undefined}
      showTabs={showTabs}
    />
  );
};
